```sql
SELECT *
FROM (
  SELECT
    team,
    player,
    score,
    ROW_NUMBER() OVER (PARTITION BY team ORDER BY score DESC) AS rank_in_team
  FROM scores
)
WHERE rank_in_team <= 3;
```
